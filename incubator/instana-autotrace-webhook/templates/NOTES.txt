  {{- if .Values.autotrace.netcore.enabled }}

  {{- end }}

    - Ruby applications
  {{- end }}
  {{- if .Values.autotrace.python.enabled }}
    - Python applications
  {{- end }}
{{- end }}
{{- if .Values.autotrace.ingress_nginx.enabled }}
